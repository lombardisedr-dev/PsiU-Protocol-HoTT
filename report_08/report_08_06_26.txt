name: PSIU Protocol Automation

on:
  push:
    branches: [ main ]
  workflow_dispatch:

permissions:
  contents: write

jobs:
  run-pipeline:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Code
        uses: actions/checkout@v4

      - name: Setup R Environment
        uses: r-lib/actions/setup-r@v2
        with:
          use-public-rspm: true

      - name: Install System Dependencies
        run: |
          Rscript -e 'install.packages("ggplot2", repos = "https://posit.co")'

      - name: Execute PsiU Validation Plot
        run: Rscript PsiU_Generate_Validation_Plot.R

      - name: Commit and Push Generated Plot
        run: |
          git config --global user.name "github-actions[bot]"
          git config --global user.email "github-actions[bot]@users.noreply.github.com"
          git add psi_u_validation_plot.png
          git commit -m "chore: aggiornato plot di validazione omotopica automatico" || echo "Nessuna modifica"
          git push origin main
